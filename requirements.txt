# ── System prerequisites (must be present on the host / Azure Web App) ───────

# Git                   (for cloning the repo)

# pnpm workspaces (monorepo) — root package.json defines the workspace.
# Lockfile: pnpm-lock.yaml  (commit this; Azure will restore from it)
